Ensure `xcodegen` is available, then write `EDGE/project.yml`:

```yaml
name: EDGE
options:
  bundleIdPrefix: com.niko.edge
  createIntermediateGroups: true
  deploymentTarget:
    iOS: "18.0"
targets:
  EDGE:
    type: application
    platform: iOS
    deploymentTarget: "18.0"
    sources:
      - path: Sources
    settings:
      base:
        SWIFT_VERSION: "5.9"
        PRODUCT_BUNDLE_IDENTIFIER: com.niko.edge.app
        MARKETING_VERSION: "1.0"
        CURRENT_PROJECT_VERSION: "1"
        TARGETED_DEVICE_FAMILY: "1"
        GENERATE_INFOPLIST_FILE: YES
        INFOPLIST_KEY_UIUserInterfaceStyle: Light
        INFOPLIST_KEY_UILaunchScreen_Generation: YES
        INFOPLIST_KEY_UIApplicationSceneManifest_Generation: YES
        INFOPLIST_KEY_CFBundleDisplayName: EDGE
        ENABLE_PREVIEWS: YES
```

`INFOPLIST_KEY_UIUserInterfaceStyle: Light` hard-locks light mode at the OS level (belt; code adds
`.preferredColorScheme(.light)` too). Do NOT hand-write a `.pbxproj` — let xcodegen own it.
